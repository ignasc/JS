Find the smallest common multiple of the provided parameters that can be evenly divided by both, as well as by all sequential numbers in the range between these parameters.

The range will be an array of two numbers that will not necessarily be in numerical order.

For example, if given 1 and 3, find the smallest common multiple of both 1 and 3 that is also evenly divisible by all numbers between 1 and 3. The answer here would be 6.

function smallestCommons(arr) {

  //PSEUDOKODAS:
  //Issiaiskinam, kuris skaicius yra didziausias is duotu dvieju masyve. Nuo to skaiciaus ir pradedam
    //Loopinam tolyn ir tikrinam sekancias salygas:
      //Tikrinam, ar tas skaicius dalus is visu, skaiciu masyve, plius visu skaiciu tarp tu dvieju masyve esanciu
        //Jei dalus, vadinasi, tai bus maziausias common multiple

  let tempStartas = Math.max(arr[0], arr[1]);
  
  while(true){ //Varom forever
    let numberFound = true;

    for(let i = Math.min(arr[0], arr[1]); i <= Math.max(arr[0], arr[1]); i++){
      if(tempStartas % i != 0){//Jei bent vienas is ju nesidalina, vadinasi i nera commonmultiple
        numberFound = false;
      }
    }

    if(numberFound){
      return tempStartas;
    }

    tempStartas++;
  }
}

smallestCommons([1,5]);
